def performance_metric2(y_true, y_predict):
    """"""
    y_pred_mean = np.mean(y_predict)
    RSS = np.sum(np.square(np.array(y_true) - np.array(y_predict)))
    TSS = np.sum(np.square(y_true - y_pred_mean))
    
    score = 1 - float(RSS)/float(TSS)

    return score

def fit_model(X, y):
    """  [X,y]"""
    
    cross_validator = KFold(n_splits = 5)
    
    regressor = DecisionTreeRegressor()

    #  [X,y]
    grid = grid.fit(X, y)

    # 
    return grid.best_estimator_
